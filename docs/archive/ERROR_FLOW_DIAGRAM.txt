WHAT ASYNCHANDLER DOES:
───────────────────────

  export const asyncHandler = (fn) => (req, res, next) => {
    // Wraps the async function
    Promise.resolve(fn(req, res, next))
      // Catches ANY promise rejection
      .catch(next);  // Passes to Express error middleware
  };
